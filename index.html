        function calculateTDEE() {

            const gender = document.getElementById('gender').value;

            const weight = parseFloat(document.getElementById('weight').value) || 60;

            const height = parseFloat(document.getElementById('height').value) || 160;

            const age = parseInt(document.getElementById('age').value) || 25;

            const activity = parseFloat(document.getElementById('activity').value);


            let bmr = (gender === 'male') 

                ? (10 * weight + 6.25 * height - 5 * age + 5)

                : (10 * weight + 6.25 * height - 5 * age - 161);


            tdee = Math.round(bmr * activity);

            document.getElementById('target-display').innerText = tdee;

            saveData();

            updateUI();

        }

        async function processImage(event) {

            const file = event.target.files[0];

            if (!file) return;


            document.getElementById('ai-loading').classList.remove('hidden');

            document.getElementById('ai-result').classList.add('hidden');


            // 壓縮圖片：為了提高成功率，將長邊限制在 800px

            const reader = new FileReader();

            reader.onload = function(e) {

                const img = new Image();

                img.onload = function() {

                    const canvas = document.createElement('canvas');

                    let width = img.width;

                    let height = img.height;

                    const maxSide = 800;

                    if (width > height && width > maxSide) { height *= maxSide / width; width = maxSide; }

                    else if (height > maxSide) { width *= maxSide / height; height = maxSide; }

                    

                    canvas.width = width;

                    canvas.height = height;

                    const ctx = canvas.getContext('2d');

                    ctx.drawImage(img, 0, 0, width, height);

                    

                    const compressedBase64 = canvas.toDataURL('image/jpeg', 0.7);

                    document.getElementById('preview-img').src = compressedBase64;

                    analyzeWithAI(compressedBase64.split(',')[1]);

                };

                img.src = e.target.result;

            };

            reader.readAsDataURL(file);

        }


        async function analyzeWithAI(base64Image) {

            const systemPrompt = "你是一個專業營養師。請分析這張食物照片，估算其食物名稱和總熱量(kcal)。請只回傳 JSON 格式：{\"food\": \"食物名稱\", \"calories\": 數字}。若無法辨識，請推測最可能的熱量。";

            

            let retries = 0;

            const maxRetries = 5;


            const callAPI = async () => {

                try {

                    const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`, {

                        method: 'POST',

                        headers: { 'Content-Type': 'application/json' },

                        body: JSON.stringify({

                            contents: [{

                                parts: [

                                    { text: "辨識圖中食物並提供卡路里" },

                                    { inlineData: { mimeType: "image/jpeg", data: base64Image } }

                                ]

                            }],

                            systemInstruction: { parts: [{ text: systemPrompt }] },

                            generationConfig: { responseMimeType: "application/json" }

                        })

                    });


                    if (!response.ok) throw new Error(`HTTP 錯誤: ${response.status}`);

                    const data = await response.json();

                    const textResult = data.candidates?.[0]?.content?.parts?.[0]?.text;

                    if (!textResult) throw new Error("AI 未能產生內容");

                    

                    const result = JSON.parse(textResult);

                    document.getElementById('ai-loading').classList.add('hidden');

                    document.getElementById('ai-result').classList.remove('hidden');

                    document.getElementById('ai-food-name').value = result.food;

                    document.getElementById('ai-food-cal').value = result.calories;


                } catch (error) {

                    if (retries < maxRetries) {

                        retries++;

                        setTimeout(callAPI, Math.pow(2, retries) * 1000);

                    } else {

                        document.getElementById('ai-loading').classList.add('hidden');

                        alert("分析失敗：" + error.message + "\n請確保光線充足或手動輸入。");

                    }

                }

            };

            callAPI();

        }


        function manualAdd(type) {

            const name = document.getElementById('manual-name').value || (type === 'food' ? '未命名飲食' : '未命名運動');

            const cal = parseInt(document.getElementById('manual-cal').value);

            if (!isNaN(cal)) {

                logs.push({ id: Date.now(), name, cal, type });

                document.getElementById('manual-name').value = '';

                document.getElementById('manual-cal').value = '';

                saveData();

                updateUI();

            } else {

                alert('請輸入熱量數值');

            }

        }


        function addAIResult() {

            const name = document.getElementById('ai-food-name').value;

            const cal = parseInt(document.getElementById('ai-food-cal').value);

            if (name && !isNaN(cal)) {

                logs.push({ id: Date.now(), name, cal, type: 'food' });

                resetAI();

                saveData();

                updateUI();

            }

        }


        function resetAI() {

            document.getElementById('ai-result').classList.add('hidden');

            document.getElementById('camera-input').value = '';

        }


        function deleteItem(id) {

            logs = logs.filter(log => log.id !== id);

            saveData();

            updateUI();

        }


        function clearAll() {

            if(confirm('確定要清空今日紀錄？')) {

                logs = [];

                saveData();

                updateUI();

            }

        }


        function updateUI() {

            const list = document.getElementById('log-list');

            list.innerHTML = '';

            let foodTotal = 0;

            let exerciseTotal = 0;


            logs.forEach(log => {

                const isFood = log.type === 'food';

                if (isFood) foodTotal += log.cal;

                else exerciseTotal += log.cal;


                const li = document.createElement('li');

                li.className = 'p-4 flex justify-between items-center group';

                li.innerHTML = `

                    <div class="flex items-center">

                        <div class="w-8 h-8 rounded-full flex items-center justify-center ${isFood ? 'bg-blue-100' : 'bg-orange-100'} mr-3">

                            ${isFood ? '🍲' : '🔥'}

                        </div>

                        <p class="font-medium text-slate-700">${log.name}</p>

                    </div>

                    <div class="flex items-center">

                        <span class="font-bold ${isFood ? 'text-blue-600' : 'text-orange-600'} mr-4">

                            ${isFood ? '+' : '-'}${log.cal}

                        </span>

                        <button onclick="deleteItem(${log.id})" class="text-slate-300 hover:text-red-500">

                            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path></svg>

                        </button>

                    </div>

                `;

                list.prepend(li);

            });


            const net = foodTotal - exerciseTotal;

            const remaining = tdee - net;

            document.getElementById('food-total').innerText = foodTotal;

            document.getElementById('exercise-total').innerText = exerciseTotal;

            document.getElementById('remaining-val').innerText = remaining;

            document.getElementById('progress-bar').style.width = Math.min((net / tdee) * 100, 100) + "%";

            if (net > tdee) document.getElementById('progress-bar').classList.replace('bg-emerald-500', 'bg-red-500');

            else document.getElementById('progress-bar').classList.replace('bg-red-500', 'bg-emerald-500');

        }

    </script>
